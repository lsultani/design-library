# Spacing & radius

The site is built on a 4px grid. Tailwind's default spacing scale
applies. Radius is **zero** everywhere that isn't a card or a chip —
squares are the brand. Cards use `2xl` (1rem) to soften. Tags use
`rounded-full` because circles read as neutral meta, not content.

export const Scale = () => {
  const steps = [0, 1, 2, 3, 4, 6, 8, 12, 16, 24, 32];
  return (
    <div style={{ display: "flex", flexDirection: "column", gap: 6, marginTop: 24 }}>
      {steps.map((s) => (
        <div key={s} style={{ display: "grid", gridTemplateColumns: "80px 80px 1fr", gap: 12, alignItems: "center" }}>
          <span style={{ fontFamily: "Space Mono, monospace", fontSize: 11, textTransform: "uppercase", letterSpacing: "0.15em" }}>{s}</span>
          <span style={{ fontFamily: "Space Mono, monospace", fontSize: 11, color: "hsl(0 0% 45%)" }}>{s * 4}px</span>
          <span style={{ height: 8, width: s * 4, background: "hsl(0 0% 6%)" }} />
        </div>
      ))}
    </div>
  );
};

export const Radius = ({ name, value, note }) => (
  <div style={{ textAlign: "center" }}>
    <div style={{ background: "hsl(0 0% 6%)", width: 72, height: 72, borderRadius: value }} />
    <div style={{ fontFamily: "Space Mono, monospace", fontSize: 10, textTransform: "uppercase", letterSpacing: "0.15em", marginTop: 8 }}>{name}</div>
    <div style={{ fontFamily: "Space Mono, monospace", fontSize: 10, color: "hsl(0 0% 45%)" }}>{value}</div>
    {note && <div style={{ fontFamily: "Space Mono, monospace", fontSize: 10, color: "hsl(0 0% 45%)" }}>{note}</div>}
  </div>
);
